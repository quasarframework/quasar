  + src\vue-directives\click-outside.js
    + New directive 'v-click-outside'
      This checks if you click outside the element where you have attached directive.
      Syntax:
        v-click-outside[:args[.id|.selectorName[.subSelectorName]]]
        where
          args: none | id | selector
          none     - it will be checked that click is outside element and his childrens
          id       - check if click if outside element and, popping up, parents until the specified id
          selector - check if click if outside element and, popping up, parents until the specified selector
      Examples:
      // simple assignment
      <div class="layout">
        <p>Sample text outside</p>
        <div v-if="showingGroup" v-click-outside="hideDivGroup">
          <p>Sample text</p>
          <p>Another Sample text</p>
        </div>
        <button @click="showDivGroup" class="pull-right">
          <i>assignment</i>
        </button>
      </div>

      // with id
      <div class="layout">
        <div class="container" id="containerid">
          <p>Sample text outside</p>
          <div
            v-if="showingGroup"
            v-click-outside:id.containerid="hideDivGroup">

            <p>Sample text</p>
            <p>Another Sample text</p>
          </div>
        </div>
        <button @click="showDivGroup" class="pull-right">
          <i>assignment</i>
        </button>
      </div>

      // with selector
      <div class="layout">
        <div class="container">
          <p>Sample text outside</p>
          <div
            v-if="showingGroup"
            v-click-outside:selector.div.container="hideDivGroup">

            <p>Sample text</p>
            <p>Another Sample text</p>
          </div>
        </div>
        <button @click="showDivGroup" class="pull-right">
          <i>assignment</i>
        </button>
      </div>

  + src\vue-directives\resize.js
    + new directive 'v-resize'
      This directive checks if the element is resized or, using arg windows
      (i.e.: v-resize:window="onResize"), the entire window and fires the
      function passed as message

  * src\utils\dom.js
    + isParentId (el, id, stopSelector)
      return true if current element or one of its parents (until stopSelector if passed) has the passed id
    + isClosest (el, selector, stopSelector)
      return true if current element or one of its parents (until stopSelector if passed) matches with passed selector
    + isSelfOrChild (el, target, stopSelector)
      return true if current element or one of its children (until stopSelector if passed) is equal to the passed target
    + isScreenMediaSize (mediaSize, paramWidth)
      return true if paramWidth or viewport().width are correspondant to mediaSize strings- like CSS media sizes
      mediaSize strings: lt-sm, sm, gt-sm, lt-md, md, gt-md, lt-bg, bg, gt-lg, lg
